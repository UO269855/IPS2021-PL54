

----------------table pedido----------------
CREATE TABLE pedido
(
IDPedido int PRIMARY KEY NOT NULL,
PrecioPedido int NULL,
fecha date NULL,
albaran varchar(25) NULL
)
----------------table orden----------------

CREATE TABLE ordenTrabajo
(
IDOrden  int PRIMARY KEY NOT NULL,
Fk_IDPedido int NOT NULL,

incidencia varchar(25) NULL,
fechaOt date NULL,
albaran varchar(25) NULL,

foreign key(Fk_IDPedido) references PUBLIC.PEDIDO(IDPedido)
)

----------------table producto----------------

CREATE TABLE producto
(
IDProducto int PRIMARY KEY NOT NULL,
nombre varchar(25)  NULL,
precio int NULL,
descripcion varchar(25) NULL,
unidades int NULL
)
----------------table productoPedido----------------

CREATE TABLE productoPedido
(
Fk_IDProducto int  NOT NULL,
Fk_IDPedido int  NOT NULL,
unidadesPedido int NULL,

foreign key(Fk_IDPedido) references PUBLIC.PEDIDO(IDPedido),
foreign key(Fk_IDProducto) references PUBLIC.PRODUCTO(IDProducto),
primary key (Fk_IDProducto,Fk_IDPedido)

)

----------------table paquete----------------

CREATE TABLE paquete
(
IDPaquete int PRIMARY KEY NOT NULL,
Fkpp_IDProducto int  NOT NULL,
Fkpp_IDPedido int  NOT NULL,


foreign key(Fkpp_IDPedido,Fkpp_IDProducto) references PUBLIC.PRODUCTOPEDIDO(Fk_IDPedido,Fk_IDProducto),

)
